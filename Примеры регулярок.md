> 	Извлечение информации из вывода команды
	Чтобы извлечь MAC-адрес, VLAN и номера портов, используем шаблон с подгруппами:
```python
import re

log_line = ('Oct  3 12:49:15.941: %SW_MATM-4-MACFLAP_NOTIF: '
            'Host f04d.a206.7fd6 in vlan 1 is flapping between port Gi0/5 and port Gi0/16')
pattern = r'Host (\S+) in vlan (\d+) is flapping between port (\S+) and port (\S+)'
match = re.search(pattern, log_line)
if match:
    mac, vlan, port1, port2 = match.groups()
    print("MAC:", mac)
    print("VLAN:", vlan)
    print("Порты:", port1, "и", port2)
```

>	 Фильтрация логов на предмет вредоносных URL
	Выбрать только те строки, где встречаются URL:
```python
import re

def filter_malicious_urls(log_filename, output_filename):
    url_pattern = r'http[s]?://(?:[a-zA-Z0-9]|[$-_@.&+]|[!*\(\),]|(?:%[0-9a-fA-F][0-9a-fA-F]))+'

    with open(log_filename, 'r', encoding='utf-8') as logfile, \
         open(output_filename, 'w', encoding='utf-8') as outfile:
        for line in logfile:
            if re.search(url_pattern, line):
                outfile.write(line)

# Пример вызова функции:
filter_malicious_urls('access.log', 'malicious_urls.txt')
```
